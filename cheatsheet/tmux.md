# main key
Ctrl b = C-b

# help
C-b ?       help
C-b /       describe key
C-b :       tmux cmd
C-Space		Cycle Layout

# panes
C-b }	    move pane right
C-b |	    split horizontally
C-b %	    split vertically
C-b x       close
C-b z       fullscreen
Alt ←       select left
Alt →       select right

# tabs (=window)
Ctrl t      open
Ctrl w      close
Alt 1-8     Goto 1-8
Alt 9       Goto last
C-b ,	    rename
